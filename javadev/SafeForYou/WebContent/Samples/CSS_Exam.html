제목 CSS Style 적용
<Options>
	// styleName : Caption에 정의된 스타일명의 스타일을 적용합니다>
	<Caption text="연간 보고서 - 2013" textAlign="center" styleName="captionStyle"/>
	// styleName : SubCaption에 정의된 스타일명의 스타일을 적용합니다.>
	<SubCaption text="RiaMore Soft" textAlign="center" styleName="subCaptionStyle"/>
</Options>
...
<Style>
	.captionStyle{fontFamily:"맑은 고딕";textDecoration:underline;fontWeight:normal;
		fontSize:14;color:#FF0000;}
	.subCaptionStyle{fontFamily:Tahoma;fontStyle:italic;fontSize:12;color:#777777;}
</Style>
